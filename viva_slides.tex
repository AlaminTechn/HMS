\begin{frame}
\frametitle{Introduction}
\begin{itemize}
\item The Hospital Management System (HMS) is designed to streamline hospital operations in Bangladesh
\item Inspired by leading institutions like Ibn Sina, Labaid, and Bardem
\end{itemize}

\textbf{Objectives:}
\begin{itemize}
\item Automate administrative processes
\item Improve patient care through digital records
\item Enhance operational efficiency
\item Ensure data security and compliance
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Bangladesh Healthcare Context}
Bangladesh's healthcare sector features modern facilities:
\begin{itemize}
\item \textbf{Ibn Sina Hospital:} Multi-specialty with cardiology, oncology, emergency services
\item \textbf{Labaid Specialized Hospital:} Known for cardiac care and cancer treatment
\item \textbf{BIRDEM:} Specialized in diabetes and endocrine disorders
\end{itemize}

Our HMS aims to emulate these institutions' digital solutions.
\end{frame}

\begin{frame}
\frametitle{System Scope}
The HMS covers:
\begin{itemize}
\item Patient Management
\item Doctor and Staff Management
\item Appointment Scheduling
\item Medical Records \& Prescriptions
\item Billing \& Payment System
\item Pharmacy Management
\item Laboratory Management
\item Ward/Room Management
\item Reporting \& Analytics
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{System Architecture}
\begin{itemize}
\item \textbf{Frontend:} React.js/Next.js (Responsive web app)
\item \textbf{Backend:} Django REST Framework (API services)
\item \textbf{Database:} PostgreSQL/MySQL (Data storage)
\item \textbf{Hosting:} Vercel (Frontend), DigitalOcean/AWS (Backend)
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Non-Functional Requirements}
\begin{itemize}
\item \textbf{Performance:} <2 seconds load time
\item \textbf{Security:} Encrypted passwords, JWT authentication
\item \textbf{Scalability:} Handle 100+ concurrent users
\item \textbf{Reliability:} 99.9\% uptime
\item \textbf{Usability:} Intuitive dashboards
\item \textbf{Compatibility:} Cross-browser, mobile responsive
\end{itemize}
\end{frame}
